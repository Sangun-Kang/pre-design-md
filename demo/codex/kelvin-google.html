kelvin.yml SOC 2 Type II controls enabled 
# Kelvin metric source configuration
workspace: scaleup-finance
warehouse:
  provider: snowflake
  database: analytics_prod
  role: kelvin_reader

metrics:
  net_revenue:
    model: marts.finance.revenue
    owner: finance-analytics
    grain: day
    dimensions:
      - region
      - segment

pricing:
  mode: per_query
  budget_alert: 5000
